VC %>%
  select(longitud, latitud, fecha, time) %>% 
  filter(year(fecha) == 2015) %>% 
  filter(month(fecha) == 1) %>%
  filter(hour(time) == 8) %>% 
  leaflet() %>%
  addTiles() %>%
  addMarkers(~longitud, ~latitud)
