sort(x1)

niveles_meses <- c(
  "Ene", "Feb", "Mar", "Abr", "May", "Jun",
  "Jul", "Ago", "Sep", "Oct", "Nov", "Dic"
)

sort(niveles_meses)
##  [1] "Abr" "Ago" "Dic" "Ene" "Feb" "Jul" "Jun" "Mar" "May" "Nov" "Oct" "Sep"
y1 <- factor(x1, levels = niveles_meses)
y1
